229  /*
230  * Returns a normal distributed random integer from a given range.
231  */
232  virtual int GetNormalIntInRange(int start, int end) {
233  if (end < start) {
234  Logger::Error(ContractFailedException("(End < Start) for normal distribution in range.\n"));
235  end = start;
236  }
237  int mean = (end + start) / 2.0f;
238  float stddev = abs(end - mean) / 2.5f;
239  std::normal_distribution<float> normalDist(mean, stddev);
240  int result = round(normalDist(generator));
241  return result < start ? start : (result > end ? end : result);
242  }
